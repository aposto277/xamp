var US = 'korochki_users', AP = 'korochki_apps', SE = 'korochki_session';
function getData(k) { return JSON.parse(localStorage.getItem(k) || '[]'); }
function setData(k, v) { localStorage.setItem(k, JSON.stringify(v)); }
function sess() { var s = sessionStorage.getItem(SE); return s ? JSON.parse(s) : null; }
function setS(v) { sessionStorage.setItem(SE, JSON.stringify(v)); }
function findUser(l) { return getData(US).find(function (u) { return u.login === l; }); }
function st(s) { return { new: 'Новая', in_progress: 'Идет обучение', completed: 'Обучение завершено' }[s] || s; }

function showErr(form, err) {
  form.querySelectorAll('.err').forEach(function (e) { e.textContent = ''; });
  for (var k in err) { var el = form.querySelector('.err[data-f="' + k + '"]'); if (el) el.textContent = err[k]; }
}

function checkReg(d) {
  var e = {};
  if (!/^[a-zA-Z0-9]{6,}$/.test(d.login)) e.login = 'Логин: латиница, от 6 символов';
  if (d.password.length < 8) e.password = 'Пароль: от 8 символов';
  if (!/^[А-Яа-яЁё\s]+$/.test(d.fullName)) e.fullName = 'ФИО: кириллица';
  if (!/^8\(\d{3}\)\d{3}-\d{2}-\d{2}$/.test(d.phone)) e.phone = 'Телефон: 8(XXX)XXX-XX-XX';
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(d.email)) e.email = 'Неверный email';
  return e;
}

function checkApp(d) {
  var e = {};
  if (!d.course) e.course = 'Выберите курс';
  if (!/^\d{2}\.\d{2}\.\d{4}$/.test(d.startDate)) e.startDate = 'Дата: ДД.ММ.ГГГГ';
  if (!d.payment) e.payment = 'Выберите оплату';
  return e;
}

document.querySelectorAll('.logout').forEach(function (a) {
  a.onclick = function (e) { e.preventDefault(); sessionStorage.removeItem(SE); location.href = 'login.html'; };
});

var page = document.body.dataset.page;

if (page === 'register') {
  document.getElementById('f').onsubmit = function (e) {
    e.preventDefault();
    var d = { login: this.login.value.trim(), password: this.password.value, fullName: this.fullName.value.trim(), phone: this.phone.value.trim(), email: this.email.value.trim() };
    var err = checkReg(d);
    showErr(this, err);
    if (Object.keys(err).length || findUser(d.login)) { if (!err.login && findUser(d.login)) showErr(this, { login: 'Логин занят' }); return; }
    var u = getData(US); u.push(d); setData(US, u);
    location.href = 'login.html';
  };
}

if (page === 'login') {
  document.getElementById('f').onsubmit = function (e) {
    e.preventDefault();
    var l = this.login.value.trim(), p = this.password.value, err = {};
    if (!l) err.login = 'Введите логин';
    if (!p) err.password = 'Введите пароль';
    showErr(this, err);
    if (Object.keys(err).length) return;
    if (l === 'Admin' && p === 'KorokNET') { setS({ login: 'Admin', role: 'admin' }); location.href = 'admin.html'; return; }
    var u = findUser(l);
    if (!u || u.password !== p) { showErr(this, { login: 'Неверный логин или пароль' }); return; }
    setS({ login: u.login, role: 'user', fullName: u.fullName });
    location.href = 'applications.html';
  };
}

if (page === 'apps') {
  var s = sess();
  if (!s || s.role !== 'user') location.href = 'login.html';
  function draw() {
    var box = document.getElementById('list'), apps = getData(AP).filter(function (a) { return a.userLogin === s.login; });
    if (!apps.length) { box.innerHTML = '<p>Нет заявок. <a href="new-application.html">Создать</a></p>'; return; }
    var h = '<table><tr><th>Курс</th><th>Дата</th><th>Статус</th><th>Отзыв</th></tr>';
    apps.forEach(function (a, i) {
      h += '<tr><td>' + a.course + '</td><td>' + a.startDate + '</td><td>' + st(a.status) + '</td><td>';
      if (a.status === 'completed') {
        h += a.review ? a.review : '<form data-i="' + i + '"><textarea required></textarea><button>OK</button></form>';
      } else h += 'После завершения курса';
      h += '</td></tr>';
    });
    box.innerHTML = h + '</table>';
    box.querySelectorAll('form').forEach(function (f) {
      f.onsubmit = function (e) {
        e.preventDefault();
        var t = this.querySelector('textarea').value.trim(), all = getData(AP), mine = all.filter(function (x) { return x.userLogin === s.login; });
        all.find(function (x) { return x.id === mine[this.dataset.i].id; }.bind(this)).review = t;
        setData(AP, all); draw();
      };
    });
  }
  draw();
}

if (page === 'newapp') {
  if (!sess() || sess().role !== 'user') location.href = 'login.html';
  document.getElementById('f').onsubmit = function (e) {
    e.preventDefault();
    var d = { course: this.course.value, startDate: this.startDate.value.trim(), payment: this.payment.value }, err = checkApp(d);
    showErr(this, err);
    if (Object.keys(err).length) return;
    var apps = getData(AP), id = apps.length ? apps[apps.length - 1].id + 1 : 1;
    apps.push({ id: id, userLogin: sess().login, course: d.course, startDate: d.startDate, payment: d.payment, status: 'new', review: '' });
    setData(AP, apps);
    location.href = 'applications.html';
  };
}

if (page === 'admin') {
  var s = sess();
  if (!s || s.role !== 'admin') location.href = 'login.html';
  function draw() {
    var stf = document.getElementById('sf').value, lf = document.getElementById('lf').value.toLowerCase();
    var apps = getData(AP).filter(function (a) {
      return (stf === 'all' || a.status === stf) && (!lf || a.userLogin.toLowerCase().indexOf(lf) >= 0);
    });
    var h = '<table><tr><th>ID</th><th>Логин</th><th>Курс</th><th>Дата</th><th>Статус</th><th></th></tr>';
    apps.forEach(function (a) {
      h += '<tr><td>' + a.id + '</td><td>' + a.userLogin + '</td><td>' + a.course + '</td><td>' + a.startDate + '</td><td>' + st(a.status) + '</td><td>';
      h += '<select data-id="' + a.id + '"><option value="new">Новая</option><option value="in_progress">Идет обучение</option><option value="completed">Обучение завершено</option></select></td></tr>';
    });
    document.getElementById('tbl').innerHTML = (apps.length ? h + '</table>' : '<p>Нет заявок</p>');
    document.querySelectorAll('select[data-id]').forEach(function (sel) {
      sel.value = apps.find(function (a) { return a.id == sel.dataset.id; }).status;
      sel.onchange = function () {
        var all = getData(AP), a = all.find(function (x) { return x.id == this.dataset.id; }.bind(this));
        if (a) a.status = this.value;
        setData(AP, all);
      };
    });
  }
  document.getElementById('sf').onchange = draw;
  document.getElementById('lf').oninput = draw;
  draw();
}

if (page === 'index') {
  var track = document.getElementById('track'), i = 0, n = track.children.length;
  function go(x) { i = (x + n) % n; track.style.transform = 'translateX(-' + i * 100 + '%)'; }
  document.getElementById('prev').onclick = function () { go(i - 1); };
  document.getElementById('next').onclick = function () { go(i + 1); };
  setInterval(function () { go(i + 1); }, 3000);
}

CREATE DATABASE IF NOT EXISTS korochki;
USE korochki;

CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  login VARCHAR(50) UNIQUE NOT NULL,
  password VARCHAR(255) NOT NULL,
  full_name VARCHAR(150) NOT NULL,
  phone VARCHAR(20) NOT NULL,
  email VARCHAR(100) NOT NULL
);

CREATE TABLE applications (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  course_name VARCHAR(200) NOT NULL,
  start_date DATE NOT NULL,
  payment_method ENUM('cash','phone') NOT NULL,
  status ENUM('new','in_progress','completed') DEFAULT 'new',
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE reviews (
  id INT AUTO_INCREMENT PRIMARY KEY,
  application_id INT NOT NULL,
  text TEXT NOT NULL,
  FOREIGN KEY (application_id) REFERENCES applications(id)
);
